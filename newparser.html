The new parser is under development!
It is under "py_parser2.py" in "lib.core".
It is available to play with within "dpl_module.py"! 

from dpl import dpl_module

dpl_module.info.LIBDIR =\
"LIBDIR HERE"

code = """
&use {std/text_io.py}
io:println("Hello, world!")
"""

processed_code = dpl_module.py_parser.process(code)
dpl_module.py_parser2.process_hlir(processed_code)
dpl_module.py_parser2.execute(processed_code)
